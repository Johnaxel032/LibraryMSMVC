CREATE PROCEDURE SP_ADD_PRODUCT
    @PRODUCT_NAME VARCHAR(80),
    @PRODUCT_DESC VARCHAR(255),
    @PRODUCT_PRICE MONEY,
    @CATEGORY_ID INT
  

AS
BEGIN
    SET NOCOUNT ON;

    INSERT INTO PRODUCT (PRODUCT_NAME, PRODUCT_DESC, PRODUCT_PRICE, PRODUCT_STATUS, CATEGORY_ID_FK)
     VALUES (@PRODUCT_NAME, @PRODUCT_DESC, @PRODUCT_PRICE, 1, @CATEGORY_ID); 
    
    DECLARE @PRODUCT_ID INT;
    SET @PRODUCT_ID = SCOPE_IDENTITY();
 


    SELECT @PRODUCT_ID AS 'PRODUCT_ID';
END
